    render: function (e) {
        var self = this;
        this.$loadingThumb.done(function (data) {
            if (e) {
                self.collection.sortByAttribute(e.target.value);
            }
            self.$el.html(self.moviesTemplate({
                movies: self.collection, 
                movieTemplate: _.template(data) 
            }));
        });
        return this;
    }
