Show Reviews view for a movie

Params
id: [Integer] ID of movie to review 
    reviews: function (id) {
        var self = this;
        this.movies.fetch({ silent: true }).then(function () {
            return self.movies.get(id).reviews.fetch({ silent: true });
        }).fail(function (xhr) {
            galactus.utils.showNotice('danger', xhr.responseText);
        }).done(function () {
            self.reviewsView = new galactus.ReviewsView({ model: self.movies.get(id) });
            $('#content').html(self.reviewsView.render().el);
        });
    }
